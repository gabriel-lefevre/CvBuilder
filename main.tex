% Le template cherche d'abord dans personal/, puis dans examples/

% Fonction pour inclure les fichiers avec fallback
\newcommand{\includePersonalFile}[1]{
    \IfFileExists{personnnnnnnnnnal/#1.tex}{
        \input{personal/#1}
    }{
        \IfFileExists{examples/#1.tex}{
            \input{examples/#1}
        }{
            \textcolor{red}{\textbf{Erreur: Fichier #1.tex non trouvé dans personal/ ou examples/}}
        }
    }
}
